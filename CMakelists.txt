#prereq
cmake_minimum_required(VERSION 3.20)
set_property(GLOBAL PROPERTY USE_FOLDERS ON)

#proj
project(gam300)

#cfg c++
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CONFIGURATION_TYPES "Debug;Release;Debug_Static_Analysis" CACHE STRING "" FORCE)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/bin)

set(CMAKE_CXX_FLAGS_DEBUG_STATIC_ANALYSIS "${CMAKE_CXX_FLAGS_DEBUG} /EHsc" CACHE STRING "" FORCE)
set(CMAKE_C_FLAGS_DEBUG_STATIC_ANALYSIS   "${CMAKE_C_FLAGS_DEBUG}" CACHE STRING "" FORCE)
set(CMAKE_EXE_LINKER_FLAGS_DEBUG_STATIC_ANALYSIS "${CMAKE_EXE_LINKER_FLAGS_DEBUG}" CACHE STRING "" FORCE)

option(ENABLE_STATIC_ANALYSIS "Enable static analysis" OFF)

#modules
set(CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/cmake")
include(import-dependencies)
include(utilities)
include(custom_tools)

#execute modules
import_dependencies()

#global compile options
#add_compile_options(/W4 /MP)
add_definitions(-D_CRT_SECURE_NO_WARNINGS)
add_definitions(-DNOMINMAX) 

if(MSVC)
add_compile_options($<$<CONFIG:Release>:/Zi>)
add_link_options($<$<CONFIG:Release>:/DEBUG>)
add_link_options($<$<CONFIG:Release>:/OPT:REF>)
add_link_options($<$<CONFIG:Release>:/OPT:ICF>)
endif()

#add this after add_definitions(-DNOMINMAX) in root CMakeLists.txt
#To enable debug symbols for release mode

# Enable exceptions for static analysis
add_compile_options($<$<OR:$<CONFIG:Debug_Static_Analysis>,$<BOOL:${ENABLE_STATIC_ANALYSIS}>>:/EHsc>)

# Try to find clang-tidy in PATH
find_program(CLANG_TIDY_EXE NAMES clang-tidy)

# If not found, try Visual Studio's bundled LLVM tools (Windows only)
if (NOT CLANG_TIDY_EXE AND MSVC)
    set(CLANG_TIDY_EXE
        "C:/Program Files/Microsoft Visual Studio/2022/Community/VC/Tools/Llvm/x64/bin/clang-tidy.exe"
    )
    if (NOT EXISTS ${CLANG_TIDY_EXE})
        message(SEND_ERROR "clang-tidy.exe not found")
    endif()

# Conditionally enable clang-tidy only in Debug builds
if (CLANG_TIDY_EXE)
    set(CMAKE_CXX_CLANG_TIDY
        "$<$<OR:$<CONFIG:Debug_Static_Analysis>,$<BOOL:${ENABLE_STATIC_ANALYSIS}>>:${CLANG_TIDY_EXE};-header-filter=${CMAKE_SOURCE_DIR}/(lib|engine|editor|tools|app)/.*;--extra-arg=/EHsc>"
    )
endif()
