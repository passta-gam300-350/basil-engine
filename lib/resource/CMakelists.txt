project(resource_lib CXX)

set(RESOURCE_CORE_DIR
    ${CMAKE_CURRENT_SOURCE_DIR}/core
)

set(RESOURCE_EXT_DIR
    ${CMAKE_CURRENT_SOURCE_DIR}/ext
)



file(GLOB_RECURSE RESOURCE_CORE_FILES RELATIVE RESOURCE_CORE_DIR
    ${CMAKE_CURRENT_LIST_DIR}/*.[ch]pp
    ${CMAKE_CURRENT_LIST_DIR}/*.h
)

file(GLOB_RECURSE RESOURCE_ADDITIONAL_EXT_FILES RELATIVE RESOURCE_EXT_DIR
    ${CMAKE_CURRENT_LIST_DIR}/*.[ch]pp
    ${CMAKE_CURRENT_LIST_DIR}/*.h
)

set(RESOURCE_EXT_DEP_LIBS
    PUBLIC assimp::assimp
)

set(RESOURCE_CORE_DEP_LIBS
)

set(RESOURCE_EXT_FILES
    RESOURCE_CORE_FILES
    RESOURCE_ADDITIONAL_EXT_FILES
)


add_library("${PROJECT_NAME}_core" STATIC ${RESOURCE_CORE_FILES})

add_library("${PROJECT_NAME}_ext" STATIC ${RESOURCE_ADDITIONAL_EXT_FILES})

target_link_libraries("${PROJECT_NAME}_ext" ${RESOURCE_EXT_DEP_LIBS})
