project(resource_lib CXX)

set(RESOURCE_CORE_DIR
    "${CMAKE_CURRENT_SOURCE_DIR}/core"
)

set(RESOURCE_EXT_DIR
    "${CMAKE_CURRENT_SOURCE_DIR}/ext"
)

file(GLOB_RECURSE RESOURCE_CORE_FILES
    ${RESOURCE_CORE_DIR}/src/*.[ch]pp
    ${RESOURCE_CORE_DIR}/src/*.h
)

file(GLOB_RECURSE RESOURCE_ADDITIONAL_EXT_FILES
    ${RESOURCE_EXT_DIR}/src/*.[ch]pp
    ${RESOURCE_EXT_DIR}/src/*.h
)

set(RESOURCE_CORE_DEP_LIBS
    INTERFACE glm
    INTERFACE tinyddsloader
    INTERFACE yaml-cpp::yaml-cpp
    INTERFACE resource-pipeline-core
)

set(RESOURCE_EXT_DEP_LIBS
    INTERFACE assimp::assimp
    INTERFACE meshoptimizer
    INTERFACE DirectXTex
    INTERFACE resource-pipeline-ext
    ${RESOURCE_CORE_DEP_LIBS}
)

set(RESOURCE_EXT_FILES
    ${RESOURCE_CORE_FILES}
    ${RESOURCE_ADDITIONAL_EXT_FILES}
)


add_library("${PROJECT_NAME}_core" INTERFACE)

add_library("${PROJECT_NAME}_ext" INTERFACE)

target_include_directories("${PROJECT_NAME}_core"
    INTERFACE "${RESOURCE_CORE_DIR}/include"
    INTERFACE "${directxtex_SOURCE_DIR}/DirectXTex"
)

target_include_directories("${PROJECT_NAME}_ext"
    INTERFACE "${RESOURCE_EXT_DIR}/include"
    INTERFACE "${RESOURCE_CORE_DIR}/include"
    INTERFACE "${directxtex_SOURCE_DIR}/DirectXTex"
)

target_link_libraries("${PROJECT_NAME}_core" ${RESOURCE_CORE_DEP_LIBS})

target_link_libraries("${PROJECT_NAME}_ext" ${RESOURCE_EXT_DEP_LIBS})

target_compile_options("${PROJECT_NAME}_core" INTERFACE
  $<$<CXX_COMPILER_ID:MSVC>:/W4>
  $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-Wall -Wextra -Wpedantic>
)


target_compile_options("${PROJECT_NAME}_ext" INTERFACE
  $<$<CXX_COMPILER_ID:MSVC>:/W4>
  $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-Wall -Wextra -Wpedantic>
)



GROUP_ENGINE_LIB_TARGETS("${PROJECT_NAME}_core")

GROUP_EDITOR_LIB_TARGETS("${PROJECT_NAME}_ext")
