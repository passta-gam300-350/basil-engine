project(graphics_lib)

)

  PUBLIC glad
  PUBLIC glfw
  PUBLIC glm
  PUBLIC assimp::assimp

  PUBLIC EnTT

 )

target_compile_definitions(${PROJECT_NAME} PUBLIC GLM_ENABLE_EXPERIMENTAL)

target_link_libraries(${PROJECT_NAME} ${ALL_LIBS})
