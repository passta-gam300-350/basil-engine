project(resource_compiler CXX)

set(RESOURCE_COMPILER_CORE_DIR
    "${CMAKE_CURRENT_SOURCE_DIR}/core"
)

set(RESOURCE_COMPILER_EXT_DIR
    "${CMAKE_CURRENT_SOURCE_DIR}/ext"
)

set(RESOURCE_COMPILER_CLI_DIR
    "${CMAKE_CURRENT_SOURCE_DIR}/app/cli"
)

set(RESOURCE_COMPILER_GUI_DIR
    "${CMAKE_CURRENT_SOURCE_DIR}/app/gui"
)

file(GLOB_RECURSE RESOURCE_COMPILER_CORE_FILES
    ${RESOURCE_COMPILER_CORE_DIR}/src/*.[ch]pp
    ${RESOURCE_COMPILER_CORE_DIR}/src/*.h
)

file(GLOB_RECURSE RESOURCE_COMPILER_ADDITIONAL_EXT_FILES
    ${RESOURCE_COMPILER_EXT_DIR}/src/*.[ch]pp
    ${RESOURCE_COMPILER_EXT_DIR}/src/*.h
)

file(GLOB_RECURSE RESOURCE_COMPILER_CLI_DRIVER_FILES
    ${RESOURCE_COMPILER_CLI_DIR}/src/*.[ch]pp
    ${RESOURCE_COMPILER_CLI_DIR}/src/*.h
)

file(GLOB_RECURSE RESOURCE_COMPILER_GUI_DRIVER_FILES
    ${RESOURCE_COMPILER_GUI_DIR}/src/*.[ch]pp
    ${RESOURCE_COMPILER_GUI_DIR}/src/*.h
)

set(RESOURCE_COMPILER_CORE_DEP_LIBS
    PUBLIC resource_lib_ext
)

set(RESOURCE_COMPILER_EXT_DEP_LIBS
    ${RESOURCE_COMPILER_CORE_DEP_LIBS}
)

set(RESOURCE_COMPILER_CLI_FILES
    ${RESOURCE_COMPILER_CORE_FILES}
    ${RESOURCE_COMPILER_CLI_DRIVER_FILES}
)

set(RESOURCE_COMPILER_GUI_FILES
    ${RESOURCE_COMPILER_CORE_FILES}
    ${RESOURCE_COMPILER_GUI_DRIVER_FILES}
    ${RESOURCE_COMPILER_ADDITIONAL_EXT_FILES}
)


add_executable("${PROJECT_NAME}_cli" ${RESOURCE_COMPILER_CLI_FILES})

add_executable("${PROJECT_NAME}_gui" WIN32 ${RESOURCE_COMPILER_GUI_FILES})

target_include_directories("${PROJECT_NAME}_cli" PUBLIC "${RESOURCE_COMPILER_CORE_DIR}/include" PUBLIC "${RESOURCE_COMPILER_CLI_DIR}/include")

target_include_directories("${PROJECT_NAME}_gui" PUBLIC "${RESOURCE_COMPILER_CORE_DIR}/include" PUBLIC "${RESOURCE_COMPILER_EXT_DIR}/include" PUBLIC "${RESOURCE_COMPILER_GUI_DIR}/include")

target_link_libraries("${PROJECT_NAME}_cli" ${RESOURCE_COMPILER_CORE_DEP_LIBS})

target_link_libraries("${PROJECT_NAME}_gui" ${RESOURCE_COMPILER_EXT_DEP_LIBS})
