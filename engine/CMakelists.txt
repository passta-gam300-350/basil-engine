project(engine CXX)
set(DOTNET_PATH_G "" CACHE STRING "Path to dotnet runtime" FORCE)
add_subdirectory(managed)

file(GLOB_RECURSE ALL_FILES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} 
    ${CMAKE_CURRENT_LIST_DIR}/*.[ch]pp
    ${CMAKE_CURRENT_LIST_DIR}/*.h
)

file(GLOB_RECURSE MANAGED_FILES_H RELATIVE ${CMAKE_CURRENT_SOURCE_DIR}/CPPGLUE
    ${CMAKE_CURRENT_LIST_DIR}/*.hpp
    ${CMAKE_CURRENT_LIST_DIR}/*.h
)

source_group("Header Files/Managed Glue" FILES ${MANAGED_FILES_H})

file(GLOB_RECURSE MANAGED_FILES_CPP RELATIVE ${CMAKE_CURRENT_SOURCE_DIR}/CPPGLUE
    ${CMAKE_CURRENT_LIST_DIR}/*.c
    ${CMAKE_CURRENT_LIST_DIR}/*.cpp
)

source_group("Source Files/Managed Glue" FILES ${MANAGED_FILES_CPP})

file(GLOB PROJECT_DIRS_HEADER RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/include/*")
foreach(dir ${PROJECT_DIRS_HEADER})
    set(dirFull "${CMAKE_CURRENT_SOURCE_DIR}/${dir}")    
    if((IS_DIRECTORY "${dirFull}"))
    # Get folder name
    set(DIR_NAME)
    get_filename_component(DIR_NAME "${dirFull}" NAME)
    file(GLOB headerFiles "${dirFull}" 
        "${dirFull}/*.h"
        "${dirFull}/*.[ch]pp"
    )
    source_group("Header Files/${DIR_NAME}" FILES ${headerFiles})

    endif()
endforeach()

file(GLOB PROJECT_DIRS_SRC RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/src/*")
foreach(dir ${PROJECT_DIRS_SRC})
    set(dirFull "${CMAKE_CURRENT_SOURCE_DIR}/${dir}")    
    if((IS_DIRECTORY "${dirFull}"))
    # Get folder name
    set(DIR_NAME)
    get_filename_component(DIR_NAME "${dirFull}" NAME)
    file(GLOB srcFiles "${dirFull}" 
        "${dirFull}/*.cpp"
        "${dirFull}/*.c"
    )
    source_group("Source Files/${DIR_NAME}" FILES ${srcFiles})

    endif()
endforeach()



set(ALL_LIBS
    PUBLIC graphics_lib
    PUBLIC scripting_lib
    PUBLIC uuid_lib
    PRIVATE spdlog
    PUBLIC EnTT
    PUBLIC tinyddsloader
    PUBLIC resource_lib_core
    PUBLIC mono_interface
    PUBLIC job_system_lib
    PUBLIC data_structure_lib
    PUBLIC Jolt
    PUBLIC fmod
    PUBLIC fmodstudio
)



add_library(${PROJECT_NAME} STATIC ${ALL_FILES})

target_include_directories(${PROJECT_NAME} 
PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include"
)

set_property(TARGET ${PROJECT_NAME} PROPERTY VS_DEBUGGER_WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}/bin")

target_compile_options(${PROJECT_NAME} PRIVATE
  $<$<CXX_COMPILER_ID:MSVC>:/W4 /MP /bigobj>
  $<$<OR:$<CONFIG:Debug_Static_Analysis>,$<BOOL:${ENABLE_STATIC_ANALYSIS}>>:/analyze>
  $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-Wall -Wextra -Wpedantic>
)

set_target_properties(${PROJECT_NAME} PROPERTIES
    VS_GLOBAL_EnableCodeAnalysisOnBuild $<$<OR:$<CONFIG:Debug_Static_Analysis>,$<BOOL:${ENABLE_STATIC_ANALYSIS}>>:true>
    VS_GLOBAL_CodeAnalysisRuleSet $<$<OR:$<CONFIG:Debug_Static_Analysis>,$<BOOL:${ENABLE_STATIC_ANALYSIS}>>:NativeRecommendedRules.ruleset>
)

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_SOURCE_DIR}/dep/vendor/fmod/api/core/lib/x64/fmod.dll" ${CMAKE_SOURCE_DIR}/bin
    COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_SOURCE_DIR}/dep/vendor/fmod/api/core/lib/x64/fmodL.dll" ${CMAKE_SOURCE_DIR}/bin
    COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_SOURCE_DIR}/dep/vendor/mono/bin/mono-2.0-sgen.dll" ${CMAKE_SOURCE_DIR}/bin
)


target_link_libraries(${PROJECT_NAME} ${ALL_LIBS})

postimport_dll(${PROJECT_NAME})
